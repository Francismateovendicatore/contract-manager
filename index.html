schema.sql — Tablas principales
-- ═══════════════════════════════════ -- ONYX REALTY — ESQUEMA POSTGRESQL -- ═══════════════════════════════════ CREATE TABLE inquilinos ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), nombre VARCHAR(120) NOT NULL, co_inquilino VARCHAR(120), cedula VARCHAR(20), telefono VARCHAR(20), email VARCHAR(80), created_at TIMESTAMPTZ DEFAULT now() ); CREATE TABLE apartamentos ( id VARCHAR(10) PRIMARY KEY, nombre VARCHAR(60) NOT NULL, direccion TEXT NOT NULL, unidad VARCHAR(60), renta_base NUMERIC(12,2) NOT NULL, estado VARCHAR(10) DEFAULT 'libre' CHECK(estado IN('ocupado','libre','mantenimiento')) ); CREATE TABLE contratos ( id VARCHAR(10) PRIMARY KEY, apt_id VARCHAR(10) REFERENCES apartamentos(id), inq_id UUID REFERENCES inquilinos(id), fecha_inicio DATE NOT NULL, renta NUMERIC(12,2) NOT NULL, deposito NUMERIC(12,2) NOT NULL, dia_pago SMALLINT CHECK(dia_pago BETWEEN 1 AND 31), notario VARCHAR(120), notas TEXT, activo BOOLEAN DEFAULT true );
schema.sql — Pagos + Índices
CREATE TABLE pagos ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), contrato_id VARCHAR(10) REFERENCES contratos(id), apt_id VARCHAR(10) REFERENCES apartamentos(id), monto NUMERIC(12,2) NOT NULL, fecha_pago DATE NOT NULL, mes SMALLINT NOT NULL, año SMALLINT NOT NULL, metodo VARCHAR(20) DEFAULT 'transferencia' CHECK(metodo IN('transferencia','efectivo','cheque')), notas TEXT, registrado_por VARCHAR(80), created_at TIMESTAMPTZ DEFAULT now() ); -- Índices para performance CREATE INDEX idx_pagos_contrato ON pagos(contrato_id); CREATE INDEX idx_pagos_apt ON pagos(apt_id); CREATE INDEX idx_pagos_periodo ON pagos(año, mes); CREATE INDEX idx_contratos_apt ON contratos(apt_id); -- Vista: estado actual por apartamento CREATE VIEW v_estado_apartamentos AS SELECT a.id, a.nombre, a.renta_base, a.estado, c.id AS contrato_id, c.renta, c.dia_pago, i.nombre AS inquilino, c.fecha_inicio, (SELECT MAX(fecha_pago) FROM pagos p WHERE p.contrato_id = c.id) AS ultimo_pago FROM apartamentos a LEFT JOIN contratos c ON c.apt_id = a.id AND c.activo LEFT JOIN inquilinos i ON i.id = c.inq_id;
queries.sql — Análisis financiero
-- ─── Ingresos mensuales ──────────────────── SELECT año, mes, COUNT(*) AS num_pagos, SUM(monto) AS total_cobrado, SUM(monto) FILTER(WHERE metodo='transferencia') AS transferencias FROM pagos GROUP BY año, mes ORDER BY año DESC, mes DESC; -- ─── Detectar morosos del mes actual ─────── SELECT c.id, a.nombre AS apartamento, i.nombre AS inquilino, c.renta AS monto_esperado, c.dia_pago FROM contratos c JOIN apartamentos a ON a.id = c.apt_id JOIN inquilinos i ON i.id = c.inq_id WHERE c.activo = true AND NOT EXISTS ( SELECT 1 FROM pagos p WHERE p.contrato_id = c.id AND p.mes = EXTRACT(MONTH FROM CURRENT_DATE) AND p.año = EXTRACT(YEAR FROM CURRENT_DATE) ); -- ─── Deuda acumulada por contrato ────────── SELECT c.id, a.nombre, i.nombre AS inquilino, (DATE_PART('year', AGE(NOW(), c.fecha_inicio)) * 12 + DATE_PART('month', AGE(NOW(), c.fecha_inicio))) * c.renta AS total_esperado, COALESCE(SUM(p.monto), 0) AS total_cobrado, ((DATE_PART('year', AGE(NOW(), c.fecha_inicio)) * 12 + DATE_PART('month', AGE(NOW(), c.fecha_inicio))) * c.renta - COALESCE(SUM(p.monto), 0)) AS deuda FROM contratos c JOIN apartamentos a ON a.id = c.apt_id JOIN inquilinos i ON i.id = c.inq_id LEFT JOIN pagos p ON p.contrato_id = c.id WHERE c.activo GROUP BY c.id, a.nombre, i.nombre, c.renta, c.fecha_inicio HAVING (DATE_PART('year',AGE(NOW(),c.fecha_inicio))*12 +DATE_PART('month',AGE(NOW(),c.fecha_inicio)))*c.renta > COALESCE(SUM(p.monto),0);

// ─── RegistrarPago ────────────────────────── async function RegistrarPago(payload) { const { contratoId, monto, fecha, metodo, mes, año, notas } = payload; await db.query(` INSERT INTO pagos (contrato_id, apt_id, monto, fecha_pago, mes, año, metodo, notas) SELECT $1, apt_id, $2, $3, $4, $5, $6, $7 FROM contratos WHERE id = $1`, [contratoId, monto, fecha, mes, año, metodo, notas]); } // ─── CalcularIngresosMensuales ────────────── async function CalcularIngresosMensuales(mes, año) { const r = await db.query(` SELECT apt_id, SUM(monto) AS total FROM pagos WHERE mes=$1 AND año=$2 GROUP BY apt_id`, [mes, año]); return r.rows; } // ─── DetectarMorosos ─────────────────────── async function DetectarMorosos(mes, año) { const activos = await db.query(` SELECT c.id, c.renta, a.nombre, i.nombre AS inquilino FROM contratos c JOIN apartamentos a ON a.id = c.apt_id JOIN inquilinos i ON i.id = c.inq_id WHERE c.activo AND NOT EXISTS ( SELECT 1 FROM pagos p WHERE p.contrato_id=c.id AND p.mes=$1 AND p.año=$2 )`, [mes, año]); return activos.rows; } // ─── CalcularDeudaPorApartamento ─────────── async function CalcularDeudaPorApartamento(aptId) { const mesesActivos = monthsSince(contrato.fechaInicio); const esperado = mesesActivos * contrato.renta; const cobrado = await totalPagadoPor(aptId); return { esperado, cobrado, deuda: esperado - cobrado }; } // ─── AnalizarSubida ──────────────────────── function AnalizarSubidaRenta(contrato, inflacionPct) { const añosContrato = yearsSince(contrato.fechaInicio); const subidaRecomendada = contrato.renta * (1 + inflacionPct / 100) ** añosContrato; return { rentaActual: contrato.renta, rentaSugerida: Math.round(subidaRecomendada/500)*500, incrementoPct: ((subidaRecomendada / contrato.renta - 1) * 100) .toFixed(1) }; }
